const TelegramBot = require('node-telegram-bot-api'); const token = 'СЮДА_ТОКЕН_БОТА'; const bot = new TelegramBot(token, { polling: true }); const WEB_APP_URL = 'https://lollipopack-webapp.vercel.app/'; console.log('Бот запущен...'); // /start + поддержка deep-link (buy_1, buy_7 и т.д.) bot.onText(/\/start(.*)/, async (msg, match) => { const chatId = msg.chat.id; const param = (match[1] || '').trim(); // Если пришли с параметром тарифа const plans = { buy_1: { days: 1, amount: 50, label: '1 день' }, buy_7: { days: 7, amount: 250, label: '7 дней' }, buy_14: { days: 14, amount: 500, label: '14 дней' }, buy_30: { days: 30, amount: 800, label: '30 дней' } }; if (plans[param]) { const plan = plans[param]; try { await bot.sendInvoice( chatId, `Подписка lollipop.cc — ${plan.label}`, `Доступ к премиум на ${plan.label}`, `premium_${plan.days}_${chatId}`, '', 'XTR', [{ label: plan.label, amount: plan.amount }] ); } catch (err) { console.log(err); bot.sendMessage(chatId, 'Ошибка при создании счёта'); } return; } // Обычный /start const text = `Привет! 👋 Добро пожаловать в *lollipop.cc* Нажми кнопку ниже, чтобы открыть приложение:`; bot.sendMessage(chatId, text, { parse_mode: 'Markdown', reply_markup: { inline_keyboard: [ [{ text: '🚀 Открыть приложение', web_app: { url: WEB_APP_URL } }], [{ text: '⭐ Оплатить подписку (Stars)', callback_data: 'choose_plan' }] ] } }); }); // Выбор тарифа внутри бота bot.on('callback_query', async (query) => { const chatId = query.message.chat.id; const data = query.data; if (data === 'choose_plan') { bot.editMessageText('Выберите срок подписки:', { chat_id: chatId, message_id: query.message.message_id, reply_markup: { inline_keyboard: [ [{ text: '1 день — 50 ⭐', callback_data: 'buy_1' }], [{ text: '7 дней — 250 ⭐', callback_data: 'buy_7' }], [{ text: '14 дней — 500 ⭐', callback_data: 'buy_14' }], [{ text: '30 дней — 800 ⭐', callback_data: 'buy_30' }] ] } }); } const plans = { buy_1: { days: 1, amount: 50, label: '1 день' }, buy_7: { days: 7, amount: 250, label: '7 дней' }, buy_14: { days: 14, amount: 500, label: '14 дней' }, buy_30: { days: 30, amount: 800, label: '30 дней' } }; if (plans[data]) { const plan = plans[data]; try { await bot.sendInvoice( chatId, `Подписка lollipop.cc — ${plan.label}`, `Доступ к премиум на ${plan.label}`, `premium_${plan.days}_${chatId}`, '', 'XTR', [{ label: plan.label, amount: plan.amount }] ); } catch (err) { bot.sendMessage(chatId, 'Ошибка при создании счёта'); } } bot.answerCallbackQuery(query.id); }); bot.on('pre_checkout_query', (q) => bot.answerPreCheckoutQuery(q.id, true)); bot.on('successful_payment', (msg) => { const chatId = msg.chat.id; bot.sendMessage(chatId, `✅ Оплата прошла успешно!\n\nПодписка активирована. Спасибо! 💜`); });
